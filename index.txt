    /*
      카드마다 오답 저장 공간을 분리합니다.
      다른 OX 카드나 이전 파일의 문항 번호가 섞이는 오류를 방지합니다.
    */
    const WRONG_STORAGE_KEY = "wrong_sports_bodybuilding_ox_v1";

    /* ==========================================================================
       📢 아래는 기존 기능 로직입니다.
       문제 풀이, 오답 저장, 오답 복습, 초기화 기능을 모두 유지합니다.
       ========================================================================== */

    let currentList = [...quizData];
    let currentIdx = 0;
    let isReviewMode = false;

    function getWrongIds() {
      const saved = localStorage.getItem(WRONG_STORAGE_KEY);
      const validIds = new Set(quizData.map(item => item.id));

      try {
        const parsed = saved ? JSON.parse(saved) : [];

        if (!Array.isArray(parsed)) {
          localStorage.removeItem(WRONG_STORAGE_KEY);
          return [];
        }

        /*
          현재 카드에 실제로 존재하는 문항 ID만 남깁니다.
          중복 ID도 함께 제거합니다.
        */
        const cleanedIds = [
          ...new Set(
            parsed.filter(id => validIds.has(id))
          )
        ];

        if (JSON.stringify(parsed) !== JSON.stringify(cleanedIds)) {
          localStorage.setItem(
            WRONG_STORAGE_KEY,
            JSON.stringify(cleanedIds)
          );
        }

        return cleanedIds;
      } catch (error) {
        localStorage.removeItem(WRONG_STORAGE_KEY);
        return [];
      }
    }

    function saveWrongId(id) {
      const wrongIds = getWrongIds();

      if (!wrongIds.includes(id)) {
        wrongIds.push(id);
        localStorage.setItem(
          WRONG_STORAGE_KEY,
          JSON.stringify(wrongIds)
        );
      }
    }

    function removeWrongId(id) {
      const wrongIds = getWrongIds().filter(
        wrongId => wrongId !== id
      );

      localStorage.setItem(
        WRONG_STORAGE_KEY,
        JSON.stringify(wrongIds)
      );
    }

    function fitQuestion() {
      const questionBox = document.querySelector(".question-box");
      const question = document.getElementById("question");

      /*
        먼저 읽기 좋은 큰 글자로 표시한 뒤,
        카드 높이를 넘는 경우에만 단계적으로 줄입니다.
        최소 크기에서도 넘치면 세로 스크롤로 읽을 수 있습니다.
      */
      const maxSize = 26;
      const minSize = 14;
      let fontSize = maxSize;

      question.style.fontSize = `${fontSize}px`;
      question.style.lineHeight = "1.48";
      questionBox.scrollTop = 0;

      while (
        fontSize > minSize &&
        question.scrollHeight > questionBox.clientHeight
      ) {
        fontSize -= 1;
        question.style.fontSize = `${fontSize}px`;
      }

      if (fontSize <= 17) {
        question.style.lineHeight = "1.38";
      }

      questionBox.scrollTop = 0;
    }

    function showQuestion() {
      const totalNum = document.getElementById("totalNum");
      const currentNum = document.getElementById("currentNum");
      const question = document.getElementById("question");
      const quizActionBox = document.getElementById("quizActionBox");
      const explainLayer = document.getElementById("explainLayer");
      const nextBtn = document.getElementById("nextBtn");

      totalNum.innerText = currentList.length;

      if (currentList.length === 0) {
        question.innerText =
          "풀 수 있는 문제가 없습니다.\n오답 노트를 확인하거나 초기화해 주세요.";
        quizActionBox.style.display = "none";
        explainLayer.style.display = "none";
        nextBtn.style.display = "none";
        currentNum.innerText = "0";
        requestAnimationFrame(fitQuestion);
        return;
      }

      quizActionBox.style.display = "flex";
      question.innerText = currentList[currentIdx].q;
      explainLayer.style.display = "none";
      explainLayer.classList.remove("compact", "extra-compact");
      nextBtn.style.display = "none";
      currentNum.innerText = currentIdx + 1;

      /*
        화면 배치가 완료된 다음 문제 글자 크기를 계산합니다.
      */
      requestAnimationFrame(fitQuestion);
    }

    function fitExplanation() {
      const explainLayer = document.getElementById("explainLayer");
      const explainText = document.getElementById("explainText");
      const length = explainText.innerText.replace(/\s/g, "").length;

      explainLayer.classList.remove("compact", "extra-compact");

      if (length >= 95) {
        explainLayer.classList.add("extra-compact");
      } else if (length >= 55) {
        explainLayer.classList.add("compact");
      }
    }

    function checkAnswer(userAns) {
      const current = currentList[currentIdx];
      const resultTitle = document.getElementById("resultTitle");
      const explainLayer = document.getElementById("explainLayer");
      const quizActionBox = document.getElementById("quizActionBox");
      const nextBtn = document.getElementById("nextBtn");

      if (userAns === current.a) {
        resultTitle.innerText = "정답입니다!";
        resultTitle.style.color = "#007cc3";

        /*
          일반 풀이와 오답 복습 모드 모두에서,
          이번에 맞힌 문제는 오답노트에서 삭제합니다.
        */
        removeWrongId(current.id);
      } else {
        resultTitle.innerText = "오답입니다.";
        resultTitle.style.color = "#e53935";

        /*
          이번에 틀린 문제는 기존 오답 여부와 관계없이
          오답노트에 추가합니다. 이미 있으면 중복 저장하지 않습니다.
        */
        saveWrongId(current.id);
      }

      document.getElementById("explainText").innerText = current.e;
      fitExplanation();
      explainLayer.style.display = "block";
      quizActionBox.style.display = "none";
      nextBtn.style.display = "block";
    }

    function nextQuestion() {
      currentIdx++;

      if (currentIdx < currentList.length) {
        showQuestion();
        return;
      }

      // 오답 복습 모드 종료 시
      if (isReviewMode) {
        const remainWrong = getWrongIds();

        // 남은 오답이 있으면 자동으로 다시 오답 복습 시작
        if (remainWrong.length > 0) {
          startReviewMode();
          return;
        }

        // 오답이 모두 없어졌을 때만 일반 모드로 복귀
        alert("🎉 모든 오답을 정복했습니다!");
      }

      alert("마지막 문제입니다!");
      resetToMain();
    }

    function startReviewMode() {
      const wrongIds = getWrongIds();

      if (wrongIds.length === 0) {
        alert("저장된 오답이 없습니다! 아주 훌륭합니다.");
        return;
      }

      currentList = quizData.filter(
        item => wrongIds.includes(item.id)
      );

      if (currentList.length === 0) {
        localStorage.removeItem(WRONG_STORAGE_KEY);
        alert("현재 카드에 해당하는 저장된 오답이 없습니다.");
        resetToMain();
        return;
      }

      currentIdx = 0;
      isReviewMode = true;

      const modeBadge = document.getElementById("modeBadge");
      modeBadge.innerText = "⚠️ 오답 복습 모드";
      modeBadge.style.background = "#e53935";

      showQuestion();
    }

    function resetToMain() {
      currentList = [...quizData];
      currentIdx = 0;
      isReviewMode = false;

      const modeBadge = document.getElementById("modeBadge");
      modeBadge.innerText = "일반 풀이 모드";
      modeBadge.style.background = "#007cc3";

      showQuestion();
    }

    function clearWrongAnswers() {
      if (
        confirm("기록된 오답 노트를 완전히 초기화하시겠습니까?")
      ) {
        localStorage.removeItem(WRONG_STORAGE_KEY);
        alert("초기화가 완료되었습니다.");
        resetToMain();
      }
    }

    window.addEventListener("resize", () => {
      requestAnimationFrame(fitQuestion);
    });

    showQuestion();
  </script>
